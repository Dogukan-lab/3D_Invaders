%%De system manager update alle sytemen binnen de ECS.
%%Als een entiteit wordt aangemaakt wordt er gekeken naar de bitset,
%%zodat alleen de benodigde systemen een nieuwe entiteit krijgen.
class SystemManager {
    std::vector<std::shared_ptr<System>> systems;

    void updateSystems();
}

    void getComponent(int componentID);
}

}
